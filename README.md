# Endpoints

# Register
curl --location --request POST '{{host}}:8001/api/user/register' \
--header 'Content-Type: application/json' \
--data-raw '{
    "name": "test", 
    "email":"test@gmail.com",
    "password":"12345678"
}'
Response:
{
    "name": "Arnold",
    "email": "test@gmail.com"
}
--------------------------------------------------------------------

# Login
curl --location --request POST '{{host}}:8001/api/user/login' \
--header 'Content-Type: application/json' \
--data-raw '{
    "email":"user@gmail.com",
    "password":"12345678"
}'
Response:

}
